아래 코드를 복사하여 데이터베이스 관리 툴(SSMS 등)에서 한 번 실행해주세요.


CREATE PROCEDURE sp_GetDynamicPayrollPivot_Fixed
    @CompanyCode NVARCHAR(10),
    @StartYear NCHAR(4),
    @EndYear NCHAR(4)
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @PivotColumns NVARCHAR(MAX);
    DECLARE @SelectColumns NVARCHAR(MAX);
    DECLARE @DynamicQuery NVARCHAR(MAX);

    -- PIVOT의 IN 절에 사용할 컬럼 목록 (지급/공제 항목)을 동적으로 생성
    SELECT @PivotColumns = STUFF(
        (
            SELECT ',' + QUOTENAME(CD)
            FROM (
                SELECT PAY_CD AS CD FROM HPOMPPD WHERE CO_CD = @CompanyCode AND LEFT(RVRS_YM, 4) BETWEEN @StartYear AND @EndYear AND PYDD_FG IN ('P2', 'P3')
                UNION
                SELECT DDCT_CD AS CD FROM HPOMPDD WHERE CO_CD = @CompanyCode AND LEFT(RVRS_YM, 4) BETWEEN @StartYear AND @EndYear AND PYDD_FG IN ('P2', 'P3')
            ) AS DistinctCodes
            ORDER BY CD
            FOR XML PATH(''), TYPE
        ).value('.', 'NVARCHAR(MAX)'), 1, 1, ''
    );

    -- 최종 SELECT 절에 사용할 컬럼 목록 ('코드_코드명' 형태의 별칭)을 동적으로 생성
    SELECT @SelectColumns = STUFF(
        (
            SELECT ',' + CHAR(13) + CHAR(10) + '    ' + QUOTENAME(Codes.CD) + ' AS ' + QUOTENAME(Codes.CD + '_' + ISNULL(CodeMaster.CTD_NM, ''))
            FROM (
                SELECT PAY_CD AS CD FROM HPOMPPD WHERE CO_CD = @CompanyCode AND LEFT(RVRS_YM, 4) BETWEEN @StartYear AND @EndYear AND PYDD_FG IN ('P2', 'P3')
                UNION
                SELECT DDCT_CD AS CD FROM HPOMPDD WHERE CO_CD = @CompanyCode AND LEFT(RVRS_YM, 4) BETWEEN @StartYear AND @EndYear AND PYDD_FG IN ('P2', 'P3')
            ) AS Codes
            LEFT JOIN SCTRL_D AS CodeMaster
                ON Codes.CD = CodeMaster.CTD_CD
                AND CodeMaster.CO_CD = @CompanyCode
                AND CodeMaster.CTRL_CD IN ('P2','P3')
            ORDER BY Codes.CD
            FOR XML PATH(''), TYPE
        ).value('.', 'NVARCHAR(MAX)'), 1, 3, ''
    );

    -- 전체 동적 PIVOT 쿼리 조립
    SET @DynamicQuery = N'
    SELECT
        EMP_CD,
        RVRS_YM,
        ' + @SelectColumns + '
    FROM (
        -- 지급 항목
        SELECT EMP_CD, RVRS_YM, PAY_CD, PAY_AM FROM HPOMPPD WHERE CO_CD = @CompanyCode AND LEFT(RVRS_YM, 4) BETWEEN @StartYear AND @EndYear AND PYDD_FG IN (''P2'', ''P3'')
        UNION ALL
        -- 공제 항목
        SELECT EMP_CD, RVRS_YM, DDCT_CD AS PAY_CD, DDCT_AM AS PAY_AM FROM HPOMPDD WHERE CO_CD = @CompanyCode AND LEFT(RVRS_YM, 4) BETWEEN @StartYear AND @EndYear AND PYDD_FG IN (''P2'', ''P3'')
    ) AS SOURCE_TABLE
    PIVOT (
        SUM(PAY_AM)
        FOR PAY_CD IN (' + @PivotColumns + ')
    ) AS PIVOT_TABLE
    ORDER BY
        RVRS_YM, EMP_CD;';

    -- 동적 쿼리 실행
    EXEC sp_executesql @DynamicQuery,
                        N'@CompanyCode NVARCHAR(10), @StartYear NCHAR(4), @EndYear NCHAR(4)',
                        @CompanyCode,
                        @StartYear,
                        @EndYear;
END;
GO
